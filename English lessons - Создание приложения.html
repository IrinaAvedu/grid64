Вот пример кода, который ты можешь использовать:

html
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>64 Grid Input</title>
  <style>
    body {
      font-family: Arial;
      padding: 20px;
    }
    textarea {
      width: 100%;
      height: 150px;
    }
    table {
      border-collapse: collapse;
      margin-top: 20px;
    }
    td {
      border: 1px solid orange;
      width: 100px;
      height: 60px;
      vertical-align: top;
      padding: 5px;
      font-size: 12px;
    }
  </style>
</head>
<body>

  <h2>Введи 64 предложения (одно на строке):</h2>
  <textarea id="inputArea" placeholder="Вводи сюда 64 предложения..."></textarea>
  <br>
  <button onclick="fillGrid()">Показать в таблице</button>

  <table id="gridTable"></table>

  <script>
    function fillGrid() {
      const input = document.getElementById("inputArea").value.trim();
      const lines = input.split('\n').map(line => line.trim()).filter(line => line !== '');
      const table = document.getElementById("gridTable");
      table.innerHTML = '';

      if (lines.length !== 64) {
        alert("Пожалуйста, введи ровно 64 предложения.");
        return;
      }

      for (let row = 0; row < 8; row++) {
        const tr = document.createElement("tr");
        for (let col = 0; col < 8; col++) {
          const td = document.createElement("td");
          td.textContent = lines[row * 8 + col];
          tr.appendChild(td);
        }
        table.appendChild(tr);
      }
    }
  </script>

</body>
</html>
